        let myChart;
        function initChart() {
            const ctx = document.getElementById('grafikRadiasi').getContext('2d');
            myChart = new Chart(ctx, {
                type: 'line',
                data: {
                    labels: [],
                    datasets: [{
                        label: 'Dose (μSv/h)', data: [],
                        borderColor: '#8B5CF6', backgroundColor: 'rgba(139, 92, 246, 0.2)',
                        borderWidth: 2, tension: 0.4, pointRadius: 0, fill: true
                    }]
                },
                options: {
                    responsive: true, maintainAspectRatio: false,
                    plugins: { legend: { display: false } }, 
                    scales: { y: { beginAtZero: true, grid: { color: 'rgba(255,255,255,0.05)' }, ticks: { color: '#888', font: {size: 10} } }, x: { grid: { display: false }, ticks: { display: false } } },
                    animation: { duration: 0 }
                }
            });
        }
        initChart();

        onValue(dataRef, (snapshot) => {
            const data = snapshot.val();
            if (data) {
                const cps = data.cps || 0; 
                const cpm = data.cpm || 0;
                const usv = parseFloat(data.dose || 0); 

                document.getElementById('nilai-cps').innerText = cps;
                document.getElementById('nilai-cpm').innerText = cpm;
                document.getElementById('nilai-usv').innerText = usv.toFixed(4);

                const kotakStatus = document.getElementById('kotak-status');
                kotakStatus.classList.remove('aman', 'waspada', 'bahaya');

                // LOGIKA DOSIS
                if (usv < 0.5) { kotakStatus.classList.add('aman'); kotakStatus.innerText = "DOSIS RENDAH"; }
                else if (usv >= 0.5 && usv < 2.0) { kotakStatus.classList.add('waspada'); kotakStatus.innerText = "DOSIS SEDANG"; }
                else { kotakStatus.classList.add('bahaya'); kotakStatus.innerText = "DOSIS TINGGI"; }

                if (myChart) {
                    const waktuSekarang = new Date().toLocaleTimeString();
                    myChart.data.labels.push(waktuSekarang);
                    myChart.data.datasets[0].data.push(usv);
                    if (myChart.data.labels.length > 20) { myChart.data.labels.shift(); myChart.data.datasets[0].data.shift(); }
                    myChart.update();
                }
            }
        });
