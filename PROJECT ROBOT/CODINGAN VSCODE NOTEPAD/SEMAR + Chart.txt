        // --- SETUP GRAFIK CHART.JS ---
        let myChart;
        function initChart() {
            const ctx = document.getElementById('grafikRadiasi').getContext('2d');
            myChart = new Chart(ctx, {
                type: 'line', // Jenis grafik garis
                data: {
                    labels: [], // Waktu (Sumbu X)
                    datasets: [{
                        label: 'Tingkat Radiasi (μSv/h)',
                        data: [], // Data Angka (Sumbu Y)
                        borderColor: '#8B5CF6', // Warna garis ungu
                        backgroundColor: 'rgba(139, 92, 246, 0.2)', // Warna area bawah garis
                        borderWidth: 3,
                        tension: 0.4, // Garis melengkung halus
                        pointRadius: 0, // Hilangkan titik biar bersih
                        fill: true
                    }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { labels: { color: 'white' } } // Warna tulisan legenda
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            grid: { color: 'rgba(255,255,255,0.1)' }, // Garis grid tipis
                            ticks: { color: '#ccc' }
                        },
                        x: {
                            grid: { display: false }, // Hilangkan grid vertikal
                            ticks: { display: false } // Sembunyikan jam biar rapi
                        }
                    },
                    animation: { duration: 0 } // Matikan animasi berat saat update data
                }
            });
        }

        // Panggil fungsi buat grafik saat halaman dimuat
        initChart();

        // --- AMBIL DATA DARI FIREBASE ---
        onValue(dataRef, (snapshot) => {
            const data = snapshot.val();
            if (data) {
                const cps = data.cps || 0; 
                const cpm = data.cpm || 0;
                const usv = parseFloat(data.dose || 0); 

                // 1. Update Kartu
                document.getElementById('nilai-cps').innerText = cps;
                document.getElementById('nilai-cpm').innerText = cpm;
                document.getElementById('nilai-usv').innerText = usv.toFixed(4);

                // 2. Update Status Bar
                const kotakStatus = document.getElementById('kotak-status');
                kotakStatus.classList.remove('aman', 'waspada', 'bahaya');

                if (usv < 0.5) {
                    kotakStatus.classList.add('aman'); kotakStatus.innerText = "KONDISI AMAN";
                } else if (usv >= 0.5 && usv < 2.0) {
                    kotakStatus.classList.add('waspada'); kotakStatus.innerText = "WASPADA!";
                } else {
                    kotakStatus.classList.add('bahaya'); kotakStatus.innerText = "BAHAYA RADIASI TINGGI!";
                }

                // 3. 🔥 UPDATE GRAFIK 🔥
                if (myChart) {
                    const waktuSekarang = new Date().toLocaleTimeString();
                    
                    // Masukkan data baru
                    myChart.data.labels.push(waktuSekarang);
                    myChart.data.datasets[0].data.push(usv);

                    // Batasi hanya 20 data terakhir biar grafik gak penuh
                    if (myChart.data.labels.length > 20) {
                        myChart.data.labels.shift(); // Hapus yang paling lama
                        myChart.data.datasets[0].data.shift();
                    }
                    myChart.update(); // Refresh grafik
                }
            }
        });

        window.keDashboard = function() {
            document.getElementById('home-view').style.display = 'none';
            document.getElementById('dashboard-view').style.display = 'block';
            document.getElementById('btn-home').classList.remove('active');
        }
